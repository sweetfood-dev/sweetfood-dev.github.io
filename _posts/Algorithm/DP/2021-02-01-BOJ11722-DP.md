**코드**
```
func solve(n : Int) {
    let arr = readLine()!.split(separator: " ").map { Int(String($0))! }
    var map = [Int](repeating:1, count: n)
    
    for i in 0 ..< arr.count {
        for j in 0 ..< i {
            if arr[i] < arr[j] {
                map[i] = max(map[i], map[j] + 1)
            }
        }
    }
    
    print(map.max()!)
}
```
